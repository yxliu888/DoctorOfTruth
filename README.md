// REST API有足够的权限设置来允许非登录用户发表评论
// 正式部署，删除这一行，在代码里使用token
add_filter('rest_allow_anonymous_comments', '__return_true');
// 默认 页数为10页，修改一个很大的也是以显示全部评论
add_filter( 'rest_comment_collection_params', function( $params ) {
	$params['per_page']['default'] = 100; // 修改默认分页大小为100
	return $params;
}, 10, 1 );
